    <script>
      (function () {
        const KEY = "theme";
        const root = document.documentElement;
        const btn = document.getElementById("themeToggle");
        const icon = document.getElementById("themeIcon");
        const label = document.getElementById("themeLabel");

        if (!btn || !icon || !label) return;

        const sunSvg =
          '<svg viewBox="0 0 24 24" width="16" height="16" fill="none" xmlns="http://www.w3.org/2000/svg"><path d="M12 18a6 6 0 1 0 0-12 6 6 0 0 0 0 12Z" stroke="currentColor" stroke-width="2"/><path d="M12 2v2M12 20v2M4.93 4.93l1.41 1.41M17.66 17.66l1.41 1.41M2 12h2M20 12h2M4.93 19.07l1.41-1.41M17.66 6.34l1.41-1.41" stroke="currentColor" stroke-width="2" stroke-linecap="round"/></svg>';

        const moonSvg =
          '<svg viewBox="0 0 24 24" width="16" height="16" fill="none" xmlns="http://www.w3.org/2000/svg"><path d="M21 13.2A7.5 7.5 0 0 1 10.8 3 6.5 6.5 0 1 0 21 13.2Z" stroke="currentColor" stroke-width="2" stroke-linejoin="round"/></svg>';

        function apply(theme, persist) {
          root.setAttribute("data-theme", theme);
          if (persist) localStorage.setItem(KEY, theme);

          const isLight = theme === "light";
          icon.innerHTML = isLight ? sunSvg : moonSvg;
          label.textContent = isLight ? "Light" : "Dark";
          btn.setAttribute(
            "aria-label",
            isLight ? "다크 모드로 전환" : "라이트 모드로 전환",
          );
        }

        const stored = localStorage.getItem(KEY);
        if (stored === "light" || stored === "dark") {
          apply(stored, false);
        } else {
          const prefersLight =
            window.matchMedia &&
            window.matchMedia("(prefers-color-scheme: light)").matches;
          apply(prefersLight ? "light" : "dark", false);

          const mq = window.matchMedia("(prefers-color-scheme: light)");
          mq.addEventListener?.("change", (e) => {
            const hasStored = localStorage.getItem(KEY);
            if (hasStored) return;
            apply(e.matches ? "light" : "dark", false);
          });
        }

        btn.addEventListener("click", () => {
          const current = root.getAttribute("data-theme") || "dark";
          const next = current === "light" ? "dark" : "light";
          apply(next, true);
          window.scrollTo({ top: 0, behavior: "smooth" });
        });
      })();
    </script>
